jQuery

■命令文：$(セレクタ).メソッド(引数);


ex.)

//要素を隠す

$('h1').hide();

//フェードアウト

$('img').fadeOut(1500);

//クリックイベント

$('#hide-text').click(function(){
$('#text').slideUp();
});

//クリックイベントとcssメソッド

$('#change-css').click(function(){
$('#text').css('color','red');
$('#text').css('font-size','50px');
});

//マウスオーバーイベント

$('#language-wrapper').hover(
function(){
$('.language-text') .fadeIn();
},
function(){
$('.language-text') .fadeOut();
}
);

ex.)

if($x % 3 == 0 && $x % 7==0){
echo "xは3の倍数かつ7の倍数です。";
}
elseif($x % 3 == 0){
echo "xは3の倍数ですが7の倍数ではありません。";

Ruby

■出力文：puts ○○


ex.)

#文字列に変数を挿入

length = 9
width = 8
area = length * width
puts "面積は#{area}です"

⇒結果：面積は72です

#文字列にインデックス付きの変数を挿入

languages = ["日本語", "英語", "スペイン語"]
puts "#{languages[1]}を話せます"

⇒結果：英語を話せます

#if文

score = 100
if score == 100
puts "満点です"
end
if score != 100
puts "満点ではありません"
end

⇒結果：満点です

#else if文

score = 73
if score > 80
puts "よくできました"
elsif score > 60
puts"まずまずです"
else
puts "がんばりましょう"
end

⇒結果：まずまずです

#変数とelse if文

year = 1994
if year < 1896
puts "#{year}年にオリンピックはまだありません"
elsif year == 1916 || year == 1940 || year == 1944
puts "#{year}年にオリンピックは開催されませんでした"
elsif year % 4 == 0
puts "#{year}年は夏季オリンピックが開催されました"
else
puts "#{year}年は夏季オリンピックイヤーではありません"
end

⇒結果：1994年は夏季オリンピックイヤーではありません

SQL

■出力文：WHERE ○○


ex.)

--「2017-08-01以前」のデータを取得

SELECT *
FROM purchases
WHERE purchased_at<="2017-08-01";

--「プリン」を含むデータを取得

SELECT *
FROM purchases
WHERE name LIKE"%プリン%";

--「にんじゃわんこ」でないデータを取得

SELECT *
FROM purchases
WHERE NOT character_name="にんじゃわんこ";

--NULLであるデータを取得

SELECT *
FROM purchases
WHERE price IS NULL;

--「食費」かつ「ひつじ仙人」のデータを取得

SELECT *
FROM purchases
WHERE category="食費" AND character_name="ひつじ仙人";

--priceカラムを基準に「降順に並び替えた」データを、最大で「5件」取得

SELECT *
FROM purchases
ORDER BY price DESC
LIMIT 5;
